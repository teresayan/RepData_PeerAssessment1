function to determine weekday or weekend

weekdayOrWeekend <- function(s) {
    switch(s, "Sunday"="weekend",
              "Monday"="weekday",
              "Tuesday"="weekday",
              "Wednesday"="weekday",
              "Thursday"="weekday", 
              "Friday"="weekday", 
              "Saturday"="weekend")
}

activity$day <- sapply(activity$date, function(x) weekdayOrWeekend(weekdays(as.Date(x))))
